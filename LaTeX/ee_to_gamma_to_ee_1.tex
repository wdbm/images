\documentclass{article}

\usepackage{feynmp}
\usepackage{feynmp-auto}

\unitlength=1.00 mm

\begin{document}

\begin{figure}

\begin{fmffile}{gg}
\begin{fmfchar*}(50,35)

\fmfleftn{i}{2}                       % 2 initial states
\fmfrightn{o}{2}                      % 4 final states

\fmf{fermion}{i1,v1}                  % e
\fmf{fermion}{v1,i2}                  % e

\fmf{photon, label=\(\gamma\)}{v1,v2} % gamma

\fmf{fermion}{o1,v2}                  % e
\fmf{fermion}{v2,o2}                  % e

\fmflabel{\(e^{-}\)}{i1}              % gamma
\fmflabel{\(e^{-}\)}{i2}              % gamma
\fmflabel{\(e^{-}\)}{o1}              % gamma
\fmflabel{\(e^{-}\)}{o2}              % gamma

\end{fmfchar*}
\end{fmffile}
\end{figure}
