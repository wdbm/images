\newcommand\particleblue[5]{% {label}{name}{mass}{charge}{spin}
    \node (#1) [particleblue]{
        \Huge${\boldsymbol{#2}}$};
    \node[below=11pt]{%
        \tiny
        \begin{tabular}{@{}p{17pt}p{43pt}@{}}
        Mass:&${#3}$\\
        Charge:&${#4}$\\
        Spin:&${#5}$
        \end{tabular}
    };
}
