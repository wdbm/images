\documentclass{standalone}
\usepackage{feynmp-auto}

\begin{document}

\begin{fmffile}{feyngraph}
    \fmfframe(5,20)(10,20){
        \begin{fmfgraph*}(150,60)
            \fmfstraight
            \fmfleft{i1,i2}
            \fmfright{o1,h,o2}
            % gluons
            \fmf{gluon}{i1,t1}
            \fmf{gluon}{t2,i2}
            \fmf{phantom,tension=0.5}{t1,o1}
            \fmf{phantom,tension=0.5}{t2,o2}
            \fmffreeze
            % t loop
            \fmf{fermion,tension=1}{t1,t2,t3,t1}
            % Higgs boson
            \fmf{dashes,tension=2.0}{t3,h}
            % labels
            \fmflabel{${g}$}{i1}
            \fmflabel{${g}$}{i2}
            \fmflabel{${H}$}{h}
        \end{fmfgraph*}
    }
\end{fmffile}

\end{document}
