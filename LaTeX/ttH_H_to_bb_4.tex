\usepackage{feynmp-auto}

\begin{document}

\begin{figure}

\begin{fmffile}{gg}
\begin{fmfchar*}(100,70)

\fmfleftn{i}{2}                         % 2 initial states
\fmfrightn{o}{4}                        % 4 final states

\fmf{curly}{i1,v1}                      % g
\fmf{curly}{i2,v2}                      % g
\fmf{fermion, label=\(\bar{t}\)}{o1,v1} % tbar
\fmf{fermion, label=\(t\)}{v1,v3}       % t
\fmf{fermion, label=\(\bar{t}\)}{v3,v2} % tbar
\fmf{fermion, label=\(t\)}{v2,o4}       % t

\fmf{dashes, label=\(H^{0}\)}{v3,v4}    % H
\fmf{fermion}{o2,v4}                    % bbar
\fmf{fermion}{v4,o3}                    % b

\fmflabel{\(g\)}{i1}                    % g
\fmflabel{\(g\)}{i2}                    % g
\fmflabel{\(\bar{t}\)}{o1}              % tbar
\fmflabel{\(\bar{b}\)}{o2}              % bbar
\fmflabel{\(b\)}{o3}                    % b
\fmflabel{\(t\)}{o4}                    % t

\end{fmfchar*}
\end{fmffile}
\end{figure}
